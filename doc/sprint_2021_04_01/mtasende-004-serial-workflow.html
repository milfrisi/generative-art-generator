def get_simulated_values(partners, cpa_inputs, date, locales=None):
    
    # Get the common data
    impressions = exps.get_impressions(spark, date, locales=locales)
    with_live_c2b_features = exps.calculate_lc2b_features(spark, impressions, date)
    common_input = exps.calculate_bapc(exps.calculate_live_c2b(spark, with_live_c2b_features, date))
    common_input = common_input.cache()
    
    first_value = True
    for partner_id in partners:
        # Filter the rows relevant to the partner
        partner_polls = impressions.where(impressions.partner_id == partner_id).select("poll_request_id").distinct()
        partner_input = common_input.join(partner_polls, 
                                               partner_polls.poll_request_id == common_input.poll_request_id,
                                               how="inner") \
                                         .drop(partner_polls.poll_request_id)
        partner_input.cache()
        
        for cpa_input in cpa_inputs:
            with_vpcs = exps.get_vpcs(partner_input, {partner_id: float(cpa_input)})
            with_vpc_priority = exps.get_vpc_priority(with_vpcs)
            with_features = exps.calculate_features(spark, with_vpc_priority, date)
            auction_input = exps.calculate_ctr(spark, with_features, date, col_name='live_ctr')

            gains = exps.calculate_gain(auction_input, beta=1)
            ad_relevances = exps.calculate_adrelevance(gains, strength=0.9)
            only_champions = exps.select_champion(ad_relevances, auction_type=exps.AuctionType.FIRST_PRICE_AUCTION, strength=0.9, beta=1)
            sorted_list = exps.sort_list(only_champions, beta=1, drop_after=15)
            predicted = exps.estimate_kpi(spark, sorted_list, date).withColumn('version', f.lit(0))  # annotate with version, it can be a string as well.
            predicted.cache()
            
            partner_agg = predicted.where(f'partner_id = {partner_id}') \
                                   .groupBy("locale") \
                                   .agg(f.sum("bapc").alias("bv_pred"), 
                                        f.sum("clicks").alias("clicks_pred"),
                                        f.sum("cpc_revenue").alias("cpc_cost"))
            
            market_agg = predicted.groupBy("locale") \
                                  .agg(f.sum("bapc").alias("market_bv_pred"), 
                                       f.sum("clicks").alias("market_clicks_pred"),
                                       f.sum("cpc_revenue").alias("market_cpc_cost")) \
                                  .withColumnRenamed("locale", "mkt_locale")

            aggregated = partner_agg.join(market_agg, partner_agg.locale == market_agg.mkt_locale, "inner") \
                                    .withColumn("partner_id", f.lit(partner_id)) \
                                    .withColumn("cpa_input_sim", f.lit(cpa_input)) \
                                    .select("partner_id", "locale", "cpa_input_sim", 
                                            "clicks_pred", "bv_pred", "cpc_cost",
                                            "market_clicks_pred", "market_bv_pred", 
                                            "market_cpc_cost")
            if first_value:
                result = aggregated
                first_value = False
            else:
                result = result.union(aggregated)
        result = result.coalesce(1000)
            
    return result
